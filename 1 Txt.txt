
def m(l,x):
  d={}
  for i in range(len(l)):
    diff = x - l[i]
    if diff in d:
      return [d[diff],i]
    else:
      d[l[i]] = i
m([1,2,3,4],6)

def m(s):
  d={}
  for i in s:
    d[i] = s.count(i)
  return d
m("xanclqbnl")

def m(a,b):
  d={}
  for i in a:
    d[i] = True
  for j in b:
    if j in d:
      return j

m([1,2,3],[2,3])
