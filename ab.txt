-- Simulating FULL OUTER JOIN in MySQL


SELECT COALESCE(s.id, t.id) AS id,
CASE WHEN s.name IS NULL THEN 'Missing in source'
	 WHEN t.name IS NULL THEN 'Missing in target'
	 WHEN s.name != t.name THEN 'Name mismatch'
	 ELSE 'Match'
     END AS status
FROM source s
LEFT JOIN target t ON s.id = t.id
WHERE s.name != t.name OR s.name IS NULL OR t.name IS NULL
UNION
SELECT COALESCE(s.id, t.id) AS id,
CASE WHEN s.name IS NULL THEN 'Missing in source'
	 WHEN t.name IS NULL THEN 'Missing in target'
	 WHEN s.name != t.name THEN 'Name mismatch'
	 ELSE 'Match'
     END AS status
FROM source s
RIGHT JOIN target t ON s.id = t.id
WHERE s.name != t.name OR s.name IS NULL OR t.name IS NULL;

sum(case when team_1=winner then 1 else 0 end) as win,
sum(case when winner='DRAW' then 1 else 0 end) as draw
from icc_world_cup2
group by team_1)
 
 union
(select team_2 as teams,count(*) as matches_played,
sum(case when team_2=winner then 1 else 0 end) as win,
sum(case when winner='DRAW' then 1 else 0 end) as draw
from icc_world_cup2
group by team_2) )

select teams,sum(matches_played) as matches_played,sum(win) as wins,sum(matches_played-win-draw) as loss,sum(draw) as draw,sum((win*2+draw*1)) as pts

order by pts desc;
