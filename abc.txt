import numpy as np

def mat_mul(a,b):
  col_a = a.shape[1]
  row_b = b.shape[0]
  if col_a != row_b:
    return "not possible"
  else:
    r = np.zeros((col_a, row_b))
    for i in range(col_a):
      for j in range(row_b):
        for k in range(col_a):
          r[i][j] = r[i][j] + a[i][k]*b[k][j]
    return r

select * from hall_events;

-- overlapping events in the same hall

with recursive cte as(
select * 
, row_number() over(order by hall_id, start_date) event_id
from hall_events)
, r_cte as(
select hall_id, start_date, end_date, event_id, 1 as flag from cte where event_id = 1
union all
select cte.hall_id, cte.start_date, cte.end_date, cte.event_id
, case when cte.hall_id = r_cte.hall_id and (cte.start_date between r_cte.start_date and r_cte.end_date
or r_cte.start_date between cte.start_date and cte.end_date) then 0 else 1 end + flag as flag
from r_cte 
inner join cte on r_cte.event_id + 1 = cte.event_id
)
select hall_id, min(start_date) start_date, max(end_date) end_date 
from r_cte
group by hall_id,flag;
