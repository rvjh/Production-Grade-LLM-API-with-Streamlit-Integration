




def m(x):
  l = [0,1]
  if x==0:
    return l[0]
  elif x==1:
    return l
  else:
    a,b = 0,1
    for i in range(2,x):
      l.append(a+b)
      a,b = b, a+b
  return l
m(10)

def per(n,r):
  if n>r:
    return m(n)/m(n-r)
def com(n,r):
  if n>r:
    return m(n)/(m(r)*m(n-r))
print(per(5,2))
print(com(5,2))

def m(x):
  if x==0 or x==1:
    return 1
  else:
    return x*m(x-1)
m(7)

def m(l,x):
  d = {}
  for i in range(len(l)):
    diff = x - l[i]
    if diff in d:
      return [d[diff],i]
    else:
      d[l[i]] = i
m([1,2,4,3],5)

def m(a,b):
  d={}
  for i in a:
    d[i] = True
  for j in b:
    if j in d:
      return j
  return None
