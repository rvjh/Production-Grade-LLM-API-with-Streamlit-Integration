

def m(x):
  if x==0 or x==1:
    return 1
  else:
    return x*m(x-1)
m(7)

def m(l,x):
  d = {}
  for i in range(len(l)):
    diff = x - l[i]
    if diff in d:
      return [d[diff],i]
    else:
      d[l[i]] = i
m([1,2,4,3],5)

def m(s):
  return s[::-1]

def m(a,b):
  d={}
  for i in a:
    d[i] = True
  for j in b:
    if j in d:
      return j
  return None

m([1,2,3],[2,3])

def m(s):
  d={}
  for i in s:
    d[i] = s.count(i)
  return d
